# 將數值切成三個Range(小、中、大)，並回傳成Factor
Class<- function(Ary){
  DATA = c()
  Min = min(Ary)
  Max = max(Ary)
  Inter = (Max - Min)/3
  
  Output <- sapply(Ary,function(x){
    if(Min<= x && x < Min+Inter){
      DATA=c(DATA,"small")
    }else if(Min+Inter<= x && x < Min+Inter*2){
      DATA=c(DATA,"median")
    }else if(Min+Inter*2<= x && x <= Min+Inter*3){
      DATA=c(DATA,"large")
    }
  })
  return(Output)
}
